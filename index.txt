<script>
        let currentLang = 'en';
        
        const GOOGLE_SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbwdiMMNJ5iaOn-E0dbXYmnx9D8dzmnSwgLSJ3R0YwIe7MFw8ZLb9Jo4rd2vYSsqm4fS/exec';

        function toggleLanguage() {
            currentLang = currentLang === 'en' ? 'es' : 'en';
            document.getElementById('lang-toggle-text').textContent = currentLang === 'en' ? 'ES' : 'EN';
            
            document.querySelectorAll('[data-en]').forEach(el => {
                const text = el.getAttribute('data-' + currentLang);
                if (!text) return;
                
                if (el.tagName === 'INPUT') {
                    el.placeholder = text;
                } else {
                    el.textContent = text;
                }
            });
        }

        function scrollToWaitlist() {
            document.getElementById('waitlist-section').scrollIntoView({ behavior: 'smooth' });
        }

        function showToast(message) {
            const toast = document.getElementById('toast');
            toast.textContent = message;
            toast.classList.add('show');
            setTimeout(() => {
                toast.classList.remove('show');
            }, 3000);
        }

        function showTherapistToast() {
            const message = currentLang === 'en' ? 
                "Applications opening soon! We'll notify our waitlist first." :
                "¡Aplicaciones abriendo pronto! Notificaremos primero a nuestra lista de espera.";
            showToast(message);
        }

        async function handleWaitlistSubmit(event) {
            event.preventDefault();
            const email = document.getElementById('email-input').value.trim();
            const btn = document.getElementById('waitlist-btn');
            
            // Validation
            if (!email || !email.includes('@') || !email.includes('.')) {
                const error = currentLang === 'en' ? 
                    'Please enter a valid email address' :
                    'Por favor ingresa un correo electrónico válido';
                showToast(error);
                return;
            }

            // Disable button and show loading
            const originalText = btn.textContent;
            btn.textContent = currentLang === 'en' ? 'Joining...' : 'Uniéndose...';
            btn.disabled = true;

            try {
                // Send to Google Apps Script
                const response = await fetch(GOOGLE_SCRIPT_URL, {
                    method: 'POST',
                    mode: 'no-cors',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({
                        email: email,
                        language: currentLang
                    })
                });

                // Success message
                const success = currentLang === 'en' ? 
                    'Welcome to kura! Check your inbox for updates.' :
                    '¡Bienvenido a kura! Revisa tu bandeja de entrada para actualizaciones.';
                showToast(success);
                document.getElementById('email-input').value = '';
                
            } catch (error) {
                console.error('Error:', error);
                const errorMsg = currentLang === 'en' ? 
                    'Something went wrong. Please try again.' :
                    'Algo salió mal. Por favor intenta de nuevo.';
                showToast(errorMsg);
            } finally {
                btn.textContent = originalText;
                btn.disabled = false;
            }
        }
    </script>
